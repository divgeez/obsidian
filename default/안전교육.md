    var ifrm = document.querySelector('[name=contents]');
    const curTime = ifrm.contentWindow.document.getElementById('curTime').innerText;
    const totalTime = ifrm.contentWindow.document.getElementById('totalTime').innerText;
    console.log("curTime="+curTime);
    console.log("totalTime="+totalTime);
    if(curTime  >= totalTime && totalTime!=="00:00") {
        setTimeout(ifrm.contentWindow.document.getElementsByClassName('icon-right-open control-icon')[0].click(), 2000);
    }
    if(ifrm.contentWindow.document.getElementById('quizStart')){ifrm.contentWindow.document.getElementById('quizStart').click();}
}, 5000);
